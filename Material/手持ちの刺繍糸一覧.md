```dataviewjs
dv.table(
  ["name", "color", "code"],
  dv.pages('"Material"')
    .where(p => p.color)
    .map(page => [
      page.file.link,
      dv.el("span", "■■■", { attr: { style: `color: ${page.color}` } }),
      page.color
    ])
);
```

```base
filters:
  and:
    - file.inFolder("Material")
    - file.hasProperty("color")

formulas:
  fileLink: "file.asLink()"
  swatch: "html(\"<span style='color: \" + color + \";'>■■■</span>\")"

properties:
  formula.fileLink:
    displayName: name
  formula.swatch:
    displayName: color
  color:
    displayName: code

views:
  - type: table
    name: Material colors
    order:
      - formula.fileLink
      - formula.swatch
      - color

```
